    def __init_subclass__(
        cls: type["Element"],
        name: str = None,
        redefined_element: "ElementType" = None,
        **kwargs: dict,
    ) -> None:
        """Customize initialization process when creating a subclass of Element.

        This class method is automatically invoked when a subclass of Element is
        created, allowing the subclass to inherit or define additional attributes
        and behaviors.

        Args:
            cls: The class object that is being initialized as a subclass.
            name: The name given to the subclass. If not provided,
                the name of the class will be used as the default.
            redefined_element: The element that this
                subclass is intending to redefine. It links back to the base element
                being redefined.
            **kwargs: Additional keyword arguments representing attributes that may be
                passed to define or redefine behaviors and properties of the subclass.

        Raises:
            TypeError: If a keyword in `kwargs` is neither found in the attributes of
                the class nor in the annotations of `ElementInterface`.

        After subclass creation, various class attributes are initialized or updated:

        - `attributes`: A dictionary of attributes, which is a copy of the existing
          attributes or an empty dictionary if it does not exist.
        - `owned_elements`: A list that tracks elements owned by this class, initialized
          as an empty list.
        - `name`: The name of the subclass, which either comes from the provided `name`
          argument or defaults to the class's `__name__` attribute.
        - `qualified_name`: The qualified name of the class, derived from `__qualname__`.
        - `owner`: Set to None, indicating that this class does not have an owner by
          default.
        - `type`: A reference to the class itself.
        - `model`: An association of the class, typically constructed through a separate
          `association` function or method.
        - `associations`: A dictionary for tracking associations, initialized as an
          empty dictionary.
        - `redefined_element`: The element set for redefinition, if provided, otherwise
          it falls back to the class's own `redefined_element` if previously set.

        Depending on the context, `__define__` or `__redefine__` method is called to
        finalize the subclass definition.

        Note:
            This method forms the crux for the creation of structured, inheritable, and
            customizable model elements within a modeling framework.

        Example:
            To create a new subclass of `Element`:

            ```python
            class MyCustomElement(Element, name='MyElement', custom_attribute='value'):
                # custom definitions and attributes here
            ```
        """
        attributes = cls.attributes = (cls.attributes or {}).copy()
        for key in kwargs:
            if key not in attributes and key not in ElementInterface.__annotations__:
                raise TypeError(
                    f"{name or cls.__name__}.__init_subclass__ got an unexpected keyword {key}"
                )
        cls.owned_elements = []
        cls.name = name or cls.__name__
        cls.qualified_name = cls.__qualname__
        cls.owner = None
        cls.type = cls
        cls.model = association(cls)
        cls.associations = {}
        cls.redefined_element = redefined_element or cls.redefined_element
        if cls.__base__.owned_elements:
            specialize(cls.__base__, cls)
        if redefined_element is None:
            cls.__define__(name=name, **kwargs)
        else:
            cls.__redefine__(redefined_element=cls.redefined_element, **kwargs)

    @classmethod
    def __redefine__(cls, **kwargs):
        for key, value in kwargs.items():
            set_attribute(cls, key, value)

    @classmethod
    def __define__(
        cls: type["Element"],
        name: str = None,
        owned_elements: Sequence["ElementType"] = (),
        **kwargs: dict,
    ) -> None:
        """Defines the class with specified attributes and set of owned elements.

        This method is responsible for setting up the attributes on the class based
        on input from the `__init_subclass__` method of the class. It allows defining
        the class according to specified rules and the addition of owned elements
        that are tied to this class.

        Args:
            cls: The class object being defined.
            name: The name of the class. If not provided, defaults to
                `None` and does not contribute to setting attributes.
            owned_elements: A sequence of element types
                that are owned by the class. These will be subsequently added to the
                class's owned elements.
            **kwargs: Additional keyword arguments representing attributes that are
                meant to be set on the class.

        For each element in the `owned_elements`, the `add_owned_element` function is
        called to register it with this class.

        It also iterates through the class annotations and existing class attributes,
        combined with `kwargs`, to set attributes on the class. Unless the key is
        specifically structured as an internal attribute (e.g., `__something__`),
        it will be set using the `set_attribute` function provided key is not part
        of `ElementInterface` annotations.
        """
        for owned_element in owned_elements:
            add_owned_element(cls, owned_element)
        for key, value in {
            **dict((name, getattr(cls, name, MISSING)) for name in cls.__annotations__),
            **cls.__dict__,
            **kwargs,
        }.items():
            if (
                f"{key[-2:]}{key[:2]}" != "____"
                and key not in ElementInterface.__annotations__
            ):
                set_attribute(cls, key, value)

    @staticmethod
    def __new__(
        cls: type["Element"],
        *args: Sequence[Any],
        **kwargs: dict,
    ) -> Union["Element", Callable[[], "Element"]]:
        """Create a new instance of an Element or its subclass.

        The `__new__` method is responsible for instance creation and is invoked
        before `__init__`. For the `Element` class, this method also initializes owned
        elements, sets up an element map for namespace management, and defines the
        model and attributes.

        Args:
            cls: The class of which an instance is to be created.
            *args: Variable-length argument list, not directly used in this method.
            **kwargs: Arbitrary keyword arguments that may include:
                - owner: an optional reference to the owner of the element instance.
                - element_map: a dictionary that holds element instances keyed by their
                  qualified names, providing a namespace for elements.

        Returns:
            A new instance of `cls` if not called on an owned element. Otherwise, it provides a lambda function that returns the instance itself when called, as a workaround to suppress further initialization logic from being executed automatically.

        The `element_map` is a crucial component that allows elements to reference
        each other correctly. When the `owner` is `None`, the hierarchy of elements is
        initialised recursively. This involves populating `attributes` with proper
        references to other elements within the map and invoking `__init__` for each
        non-root element with any additional keyword arguments matching their qualified
        name. The `element_map` should initially include at least the current instance
        with its qualified name.

        """
        self = super().__new__(cls)
        self.owner = kwargs.pop("owner", None)
        self.owned_elements = []
        self.attributes = {}
        self.model = self.owner.model if self.owner is not None else self
        element_map = kwargs.pop(
            "element_map", {self.qualified_name: self}
        )  # create a namespace for the elements in the element
        for owned_element in cls.owned_elements:
            instance = owned_element(
                owner=self,
                element_map=element_map,
            )()  # using the extra function call to prevent __init__ from being called
            element_map[owned_element.qualified_name] = instance
            self.owned_elements.append(instance)
        if self.owner is None:
            for element in reversed(element_map.values()):
                for name, value in element.__class__.attributes.items():
                    if is_element(value):
                        value = element.attributes[name] = element_map[
                            value.qualified_name
                        ]
                        setattr(element, str(name), value)
                if element is not self:
                    element.__init__(**kwargs.pop(element.qualified_name, {}))
            # this is the root element of the element, so we can start initializing
            return self
        # a hack to prevent __init__ from being called
        return lambda _self=self: _self
